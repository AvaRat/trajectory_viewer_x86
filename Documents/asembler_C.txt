global f	-> funkcja ta bedzie widoczna na zewnątrz


f:
	push EBP
	mov EBP, ESP
	mov EAX, [EBP+8]	-> []=dereferencja	-> pobieramy argument ze stosu(8 B wyżej)
	
begin:
	mov cl, [EAX]
	cmp cl, 0	-> compare and save flags	// patrzeć w dokumentacji, czy dana flaga na pewno jest używana przez dany skok

	add cl, 1
	mov [EAX], cl
	inc EAX
	jmp begin
end:
